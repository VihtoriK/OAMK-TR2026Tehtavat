a) Tulosta DVD-elokuvien kielet (language) aakkosjärjestyksessä
SELECT * FROM language ORDER BY name;

b) Tulosta kaikkien niiden näyttelijöiden elokuvat, joiden sukunimi on Temple.
SELECT first_name, last_name, title FROM actor JOIN film_actor ON actor.actor_id = film_actor.actor_id JOIN film ON film.film_id = film_actor.film_id WHERE last_name = 'Temple'; 

c) Tulosta elokuvassa "Ghost Groundhog" näytelleet näyttelijät.
SELECT first_name, last_name, title FROM actor JOIN film_actor ON actor.actor_id = film_actor.actor_id JOIN film ON film.film_id = film_actor.film_id WHERE title = 'Ghost Groundhog';

d) Montako kauhuelokuvaa ("Horror") tietokannassa on ?
SELECT COUNT(*) FROM category c JOIN film_category fc ON c.category_id = fc.category_id WHERE name='Horror';

e) Tulosta kaikki kauhuelokuvat.
SELECT title FROM category c JOIN film_category fc ON c.category_id = fc.category_id JOIN film f ON f.film_id = fc.film_id WHERE name='Horror';
